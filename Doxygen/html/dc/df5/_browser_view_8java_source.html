259  // Inner class to deal with link-clicks and mouse-overs Mostly taken from

261  private class LinkListener implements ChangeListener<State> {
262  public static final String EVENT_CLICK = "click";
263  public static final String EVENT_MOUSEOVER = "mouseover";
264  public static final String EVENT_MOUSEOUT = "mouseout";
265 
266  @Override
267  public void changed (ObservableValue<? extends State> ov, State oldState, State newState) {
268  if (newState == Worker.State.SUCCEEDED) {
269  EventListener listener = event -> {
270  final String href = ((Element)event.getTarget()).getAttribute("href");
271  if (href != null) {
272  String domEventType = event.getType();
273  if (domEventType.equals(EVENT_CLICK)) {
274  showPage(href);
275  } else if (domEventType.equals(EVENT_MOUSEOVER)) {
276  showStatus(href);
277  } else if (domEventType.equals(EVENT_MOUSEOUT)) {
278  showStatus(BLANK);
279  }
280  }
281  };
282  Document doc = myPage.getEngine().getDocument();
283  NodeList nodes = doc.getElementsByTagName("a");
284  for (int i = 0; i < nodes.getLength(); i++) {
285  EventTarget node = (EventTarget)nodes.item(i);
286  node.addEventListener(EVENT_CLICK, listener, false);
287  node.addEventListener(EVENT_MOUSEOVER, listener, false);
288  node.addEventListener(EVENT_MOUSEOUT, listener, false);
289  }
290  }
291  }
